PARTE 1: DESCRIPCION GENERAL DEL PROYECTO
================================================================================

NOMBRE: Neo-App
DESCRIPCION: Aplicación web para generar historias narrativas visuales. Los
usuarios suben un guion maestro, el sistema analiza el texto con LLM para
detectar personajes/locaciones/props, permite gestión manual de personajes con
imágenes, genera un Story Bible completo, crea Scene Breakdown con prompts
optimizados para generación de video, y exporta en múltiples formatos.

OBJETIVO: Ser lo más parecida posible a https://magiclight.ai/

STACK TECNOLOGICO:
- Frontend: React 19 + Vite + TypeScript + Tailwind CSS 4 + shadcn/ui
- Backend: Node.js + Express + TypeScript
- Base de datos: MySQL/TiDB (Drizzle ORM)
- LLM: OpenAI-compatible API (OPENAI_API_KEY)
- Autenticación: Manus OAuth (integrado)
- Almacenamiento: S3 (para imágenes)
- Herramientas: pnpm, tsx, Vitest

COMPATIBILIDAD:
- OpenCode: Sí (Node.js + React)
- Antigravity: Sí (Node.js + React)
- Localhost: Sí (npm run dev)
- Producción: Sí (npm run build + npm start)

import { OpenAI } from "openai";

interface StoryBibleResponse {
  characters: Array<{ name: string; description: string }>;
  locations: Array<{ name: string; description: string; timeOfDay: string }>;
  props: Array<{ name: string; description: string }>;
}

interface SceneBreakdownResponse {
  scenes: Array<{
    sceneNumber: number;
    time: string;
    place: string;
    beat: string;
    grokVideoPrompt: string;
    negative: string;
  }>;
}

const NARRATIVE_DIRECTOR_PROMPT = `Eres un Director Narrativo y Prompt Engineer especializado en cuentos infantiles (3–7 años) para generar vídeo por clips en Grok. Tu misión: tomar un GUIÓN/TRANSCRIPT y convertirlo en un plan completo de escenas consecutivas, cada una de 10 segundos, y crear un prompt de vídeo (Text-to-Video) para cada escena.

IDIOMA DEL CUENTO: Español (neutro).
PROMPTS DE VÍDEO: En INGLÉS (para mejorar resultados del modelo).
TONO: calmado, tierno, para dormir. Cero miedo intenso. Cero violencia explícita.
DURACIÓN POR ESCENA: 10 segundos EXACTOS (si una parte dura más, divídela en varias escenas de 10s).
FORMATO DE SALIDA: SOLO texto plano (sin Markdown, sin viñetas).

PLANTILLA EXACTA POR ESCENA
Scene01
TIME: 00:00–00:10
PLACE: EXT.-
BEAT: ____
GROK_VIDEO_PROMPT: ____
NEGATIVE: ____

CONTROL FINAL ANTES DE ENTREGAR
¿Todas las escenas duran 10s y cubren todo el cuento?
¿Personajes y vestuario consistentes?
¿Sin texto en pantalla, sin watermark, sin violencia?
¿Prompts en inglés, cuento en español solo en BEAT si hace falta?`;

export class LLMService {
  private client: OpenAI;

  constructor(apiKey: string, apiUrl: string) {
    this.client = new OpenAI({
      apiKey,
      baseURL: apiUrl,
    });
  }

  async analyzeScript(scriptContent: string): Promise<{ storyBible: StoryBibleResponse; sceneBreakdown: SceneBreakdownResponse }> {
    try {
      const storyBibleResponse = await this.client.chat.completions.create({
        model: process.env.LLM_MODEL || "gpt-4.1-mini",
        messages: [
          {
            role: "system",
            content: "Eres un experto en análisis narrativo. Extrae personajes, locaciones y props del guion proporcionado. Responde en JSON válido.",
          },
          {
            role: "user",
            content: `Analiza este guion y extrae personajes, locaciones y props:\n\n${scriptContent}\n\nResponde en este formato JSON:\n{"characters": [{"name": "...", "description": "..."}], "locations": [{"name": "...", "description": "...", "timeOfDay": "..."}], "props": [{"name": "...", "description": "..."}]}`,
          },
        ],
        response_format: { type: "json_object" },
      });

      const storyBible = JSON.parse(storyBibleResponse.choices[0]?.message?.content || "{}") as StoryBibleResponse;

      const sceneBreakdownResponse = await this.client.chat.completions.create({
        model: process.env.LLM_MODEL || "gpt-4.1-mini",
        messages: [
          {
            role: "system",
            content: NARRATIVE_DIRECTOR_PROMPT,
          },
          {
            role: "user",
            content: `Genera el Scene Breakdown para este guion:\n\n${scriptContent}`,
          },
        ],
      });

      const sceneBreakdownText = sceneBreakdownResponse.choices[0]?.message?.content || "";
      const sceneBreakdown = this.parseSceneBreakdown(sceneBreakdownText);

      return { storyBible, sceneBreakdown };
    } catch (error) {
      console.error("Error en LLM:", error);
      throw error;
    }
  }

  private parseSceneBreakdown(text: string): SceneBreakdownResponse {
    const scenes: SceneBreakdownResponse["scenes"] = [];
    const sceneMatches = text.match(/Scene\d+[\s\S]*?(?=Scene\d+|$)/g) || [];

    sceneMatches.forEach((sceneText, index) => {
      const timeMatch = sceneText.match(/TIME:\s*([^\n]+)/);
      const placeMatch = sceneText.match(/PLACE:\s*([^\n]+)/);
      const beatMatch = sceneText.match(/BEAT:\s*([^\n]+)/);
      const promptMatch = sceneText.match(/GROK_VIDEO_PROMPT:\s*([\s\S]*?)(?=NEGATIVE:|$)/);
      const negativeMatch = sceneText.match(/NEGATIVE:\s*([^\n]+)/);

      scenes.push({
        sceneNumber: index + 1,
        time: timeMatch?.[1]?.trim() || `00:${String(index * 10).padStart(2, "0")}–00:${String((index + 1) * 10).padStart(2, "0")}`,
        place: placeMatch?.[1]?.trim() || "INT. UNKNOWN",
        beat: beatMatch?.[1]?.trim() || "",
        grokVideoPrompt: promptMatch?.[1]?.trim() || "",
        negative: negativeMatch?.[1]?.trim() || "",
      });
    });

    return { scenes };
  }
}

export class ImageService {
  async generateImage(prompt: string, negativePrompt: string): Promise<{ url: string }> {
    try {
      const response = await fetch(`${process.env.BUILT_IN_FORGE_API_URL}/api/image/generate`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${process.env.BUILT_IN_FORGE_API_KEY}`,
        },
        body: JSON.stringify({
          prompt,
          negativePrompt,
          steps: 20,
          guidanceScale: 7.5,
        }),
      });

      if (!response.ok) throw new Error("Image generation failed");
      const data = await response.json() as { url: string };
      return data;
    } catch (error) {
      console.error("Error generating image:", error);
      throw error;
    }
  }
}

import { z } from "zod";
import { protectedProcedure, publicProcedure, router } from "./_core/trpc.js";
import { getDb } from "../db.js";
import { projects } from "../../drizzle/schema.js";
import { eq } from "drizzle-orm";

export const projectsRouter = router({
  list: protectedProcedure.query(async ({ ctx }) => {
    const db = await getDb();
    if (!db) return [];
    const userProjects = await db.select().from(projects).where(eq(projects.userId, ctx.user.id));
    return userProjects;
  }),

  create: protectedProcedure
    .input(z.object({ title: z.string(), description: z.string().optional() }))
    .mutation(async ({ ctx, input }) => {
      const db = await getDb();
      if (!db) throw new Error("Database not available");
      const result = await db.insert(projects).values({
        userId: ctx.user.id,
        title: input.title,
        description: input.description,
        status: "draft",
      });
      return { id: result.insertId, ...input, status: "draft" };
    }),

  get: protectedProcedure
    .input(z.object({ id: z.number() }))
    .query(async ({ ctx, input }) => {
      const db = await getDb();
      if (!db) return null;
      const project = await db.select().from(projects).where(eq(projects.id, input.id)).limit(1);
      return project[0] || null;
    }),

  update: protectedProcedure
    .input(z.object({ id: z.number(), title: z.string().optional(), description: z.string().optional(), status: z.enum(["draft", "analyzing", "completed"]).optional() }))
    .mutation(async ({ ctx, input }) => {
      const db = await getDb();
      if (!db) throw new Error("Database not available");
      await db.update(projects).set({ title: input.title, description: input.description, status: input.status }).where(eq(projects.id, input.id));
      return { success: true };
    }),

  delete: protectedProcedure
    .input(z.object({ id: z.number() }))
    .mutation(async ({ ctx, input }) => {
      const db = await getDb();
      if (!db) throw new Error("Database not available");
      await db.delete(projects).where(eq(projects.id, input.id));
      return { success: true };
    }),
});
